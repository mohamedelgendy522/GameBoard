153bool X_O_Shift::is_win(Player<char>* player) {
154 const char sym = player->get_symbol();
155
156 // Check sliding 4-length matches
157 auto all_equal_4 = [&](char a, char b, char c, char d) {
158 if (a==b && b==c && a!=blank_symbol && a == sym) return true;
159 if (b==c && c==d && d!=blank_symbol && b == sym) return true;
160 return false;
161 };
162
163 // Check simple 3-length diagonal matches
164 auto all_equal_3 = [&](char a, char b, char c) {
165 return a == b && b == c && a != blank_symbol && a == sym;
166 };
167
168 // Check rows and columns
169 for (int i = 0; i < rows; ++i) {
170 if (all_equal_4(board[i][0], board[i][1], board[i][2], board[i][3]) ||
171 all_equal_4(board[0][i], board[1][i], board[2][i], board[3][i]))
172 return true;
173 }
174
175 // 3-length Diagonals (upper)
176 if ((all_equal_3(board[2][0], board[1][1], board[0][2]) && board[0][2] == sym) ||
177 (all_equal_3(board[3][1], board[2][2], board[1][3]) && board[3][1] == sym))
178 return true;
179
180 // 3-length Diagonals (lower)
181 if ((all_equal_3(board[1][0], board[2][1], board[3][2]) && board[1][0] == sym) ||
182 (all_equal_3(board[0][1], board[1][2], board[2][3]) && board[0][1] == sym))
183 return true;
184
185 // Full 4-length diagonals
186 if (all_equal_4(board[0][0],board[1][1],board[2][2],board[3][3]) ||
187 all_equal_4(board[0][3],board[1][2],board[2][1],board[3][0]))
188 return true;
189
190 return false;
