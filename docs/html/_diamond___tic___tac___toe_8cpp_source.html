64bool Diamond_X_O_Board::is_win(Player<char>* player) {
65 const char sym = player->get_symbol();
66
67 vector<pair<int, int>> directions = {
68 {0, 1}, // Horizontal
69 {1, 0}, // Vertical
70 {1, 1}, // Diagonal ↘
71 {1, -1} // Diagonal ↙
72 };
73
74 vector<bool> found_three(4, false);
75 vector<bool> found_four(4, false);
76
77 for (int i = 0; i < rows; ++i) {
78 for (int j = 0; j < columns; ++j) {
79 if (!is_valid_position(i, j) || board[i][j] != sym)
80 continue;
81
82 for (int dir_idx = 0; dir_idx < 4; ++dir_idx) {
83 auto& dir = directions[dir_idx];
84 int dx = dir.first, dy = dir.second;
85
86 // Check 3 in line
87 bool valid_three = true;
88 for (int k = 0; k < 3; ++k) {
89 int ni = i + k*dx, nj = j + k*dy;
90 if (!is_valid_position(ni, nj) || board[ni][nj] != sym) {
91 valid_three = false;
92 break;
93 }
94 }
95 if (valid_three) found_three[dir_idx] = true;
96
97 // Check 4 in line
98 bool valid_four = true;
99 for (int k = 0; k < 4; ++k) {
100 int ni = i + k*dx, nj = j + k*dy;
101 if (!is_valid_position(ni, nj) || board[ni][nj] != sym) {
102 valid_four = false;
103 break;
104 }
105 }
106 if (valid_four) found_four[dir_idx] = true;
107 }
108 }
109 }
110
111 for (int i = 0; i < 4; ++i) {
112 for (int j = 0; j < 4; ++j) {
113 if (i != j && found_three[i] && found_four[j]) {
114 return true;
115 }
116 }
117 }
118
119 return false;
120}

211int Diamond_X_O_Board::evaluate() {
212 if(is_win(computer_player)) return 100;
213 if(is_win(human_player)) return -100;
214
215 int score = 0;
216 score += count_lines(computer_player, 2) * 5;
217 score -= count_lines(human_player, 2) * 5;
218 score += count_lines(computer_player, 3) * 20;
219 score -= count_lines(human_player, 3) * 20;
220
221 // Center control (3,3 is the new center)
222 if (board[3][3] == computer_player->get_symbol()) score += 10;
223

230Move<char>* Diamond_X_O_Board::find_best_move() {
231 int best_score = -1000;
232 Move<char>* best_move = nullptr;
233 vector<pair<int, int>> valid_positions = get_valid_positions();
234 int search_depth = 5;
235
236 for (auto& pos : valid_positions) {
237 int i = pos.first, j = pos.second;
238 board[i][j] = computer_player->get_symbol();
239 n_moves++;
240
241 int score = minimax(false, -1000, 1000, search_depth);
242
243 board[i][j] = blank_symbol;
244 n_moves--;
245
246 if (score > best_score) {
247 best_score = score;
248 best_move = new Move<char>(i, j, computer_player->get_symbol());
249 }
250 }
251
252 if (best_move == nullptr && !valid_positions.empty()) {
253 auto& first = valid_positions[0];
254 best_move = new Move<char>(first.first, first.second, computer_player->get_symbol());
255 }
256 return best_move;

262int Diamond_X_O_Board::minimax(bool is_maximizing, int alpha, int beta, int depth) {
263 if (is_terminal() || depth == 0) return evaluate();
264
265 vector<pair<int, int>> valid_positions = get_valid_positions();
266
267 if (is_maximizing) {
268 int max_eval = -1000;
269 for (auto& pos : valid_positions) {
270 int i = pos.first, j = pos.second;
271 board[i][j] = computer_player->get_symbol();
272 n_moves++;
273 int eval = minimax(false, alpha, beta, depth - 1);
274 board[i][j] = blank_symbol;
275 n_moves--;
276 max_eval = max(max_eval, eval);
277 alpha = max(alpha, eval);
278 if (beta <= alpha) break;
279 }
280 return max_eval;
281 }
282 else {
283 int min_eval = 1000;
284 for (auto& pos : valid_positions) {
285 int i = pos.first, j = pos.second;
286 board[i][j] = human_player->get_symbol();
287 n_moves++;
288 int eval = minimax(true, alpha, beta, depth - 1);
289 board[i][j] = blank_symbol;
290 n_moves--;
291 min_eval = min(min_eval, eval);
292 beta = min(beta, eval);
293 if (beta <= alpha) break;
294 }
295 return min_eval;
296 }
