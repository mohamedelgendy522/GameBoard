40bool Memory_Board::update_board(Move<char>* move) {
41 int x = move->get_x();
42 int y = move->get_y();
43 char mark = move->get_symbol();
44
45 // Validate move and apply if valid
46 if (!(x < 0 || x >= rows || y < 0 || y >= columns) &&
47 (board[x][y] == blank_symbol || mark == 0)) {
48
49 if (mark == 0) { // Undo move
50 n_moves--;
51 board[x][y] = blank_symbol;
52 }
53 else { // Apply move
54 n_moves++;
55 board[x][y] = toupper(mark);
56 }
57 return true;
58 }
59 return false;
60}

68bool Memory_Board::is_win(Player<char>* player) {
69 const char sym = player->get_symbol();
70
71 auto all_equal = [&](char a, char b, char c) {
72 return a == b && b == c && a != blank_symbol;
73 };
74
75 // Check rows and columns
76 for (int i = 0; i < rows; ++i) {
77 if ((all_equal(board[i][0], board[i][1], board[i][2]) && board[i][0] == sym) ||
78 (all_equal(board[0][i], board[1][i], board[2][i]) && board[0][i] == sym))
79 return true;
80 }
81
82 // Check diagonals
83 if ((all_equal(board[0][0], board[1][1], board[2][2]) && board[1][1] == sym) ||
84 (all_equal(board[0][2], board[1][1], board[2][0]) && board[1][1] == sym))
85 return true;
86
87 return false;
88}
