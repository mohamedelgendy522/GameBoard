72bool Word_Board::update_board(Move<char>* move) {
73 int x = move->get_x();
74 int y = move->get_y();
75 char val = move->get_symbol();
76
77 // Bounds check
78 if (x < 0 || x >= rows || y < 0 || y >= columns) {
79 cout << "\n Invalid move! Position (" << x << ", " << y << ") is outside the board.\n";
80 return false;
81 }
82
83 // Occupied cell check
84 if (board[x][y] != '.') {
85 cout << "\n Invalid move! Cell (" << x << ", " << y << ") is already filled.\n";
86 return false;
87 }
88
89 board[x][y] = val;
90 n_moves++;
91
92 return true;
93}

106bool Word_Board::is_win(Player<char>* player) {
107 auto ValidWord = [&](char a, char b, char c) -> string {
108 string s;
109 s.push_back(a);
110 s.push_back(b);
111 s.push_back(c);
112 return s;
113 };
114
115 // Check rows and columns
116 for (int i = 0; i < rows; ++i) {
117 string rowWord = ValidWord(board[i][0], board[i][1], board[i][2]);
118 if (dict.count(rowWord)) return true;
119
120 string colWord = ValidWord(board[0][i], board[1][i], board[2][i]);
121 if (dict.count(colWord)) return true;
122 }
123
124 // Check diagonals
125 string d1 = ValidWord(board[0][0], board[1][1], board[2][2]);
126 if (dict.count(d1)) return true;
127
128 string d2 = ValidWord(board[0][2], board[1][1], board[2][0]);
129 if (dict.count(d2)) return true;
130
131 return false;
132}

210Move<char>* Word_UI::get_move(Player<char>* player) {
211 int x, y;
212 char val;
213
214 // HUMAN MOVE
215 if (player->get_type() == PlayerType::HUMAN) {
216 cout << "\n" << player->get_name() << ", please enter your move (x,y,char): ";
217 cin >> x >> y >> val;
218 val = toupper(val);
219 }
220
221 else { // COMPUTER MOVE
222 Word_Board* board = dynamic_cast<Word_Board*>(player->get_board_ptr());
223 bool found = false;
224
225 // Try to find a winning move
226 for (int i = 0; i < board->get_rows() && !found; ++i) {
227 for (int j = 0; j < board->get_columns() && !found; ++j) {
228 if (board->get_cell(i, j) == '.') {
229 for (char c = 'A'; c <= 'Z' && !found; ++c) {
230
231 Word_Board tempBoard = *board;
232
233 Move<char> tempMove(i, j, c);
234 tempBoard.update_board(&tempMove);
235
236 if (tempBoard.is_win(nullptr)) {
237 x = i; y = j; val = c;
238 found = true;
239 }
240 }
241 }
242 }
243 }
244
245 // No winning move found → play random
246 if (!found) {
247 for (int attempts = 0; attempts < 50 && !found; ++attempts) {
248 x = rand() % 3;
249 y = rand() % 3;
250 if (board->get_cell(x, y) == '.') {
251 val = 'A' + (rand() % 26);
252 found = true;
253 }
254 }
255
256 // Safety fallback
257 if (!found) {
258 for (int i = 0; i < 3 && !found; i++)
259 for (int j = 0; j < 3 && !found; j++)
260 if (board->get_cell(i, j) == '.') {
261 x = i; y = j;
262 val = 'A' + (rand() % 26);
263 found = true;
264 }
265 }
266 }
267
268 cout << "Computer chooses: " << x << " " << y << " " << val << endl;
269 }
270
271 return new Move<char>(x, y, val);
272}
