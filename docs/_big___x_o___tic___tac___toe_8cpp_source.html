84int Big_XO_Board::count_sequences(char sym) {

86 int count = 0;
87
88 // Rows
89 for (int r = 0; r < rows; ++r)
90 for (int c = 0; c <= columns - 3; ++c)
91 if (board[r][c] == sym && board[r][c + 1] == sym && board[r][c + 2] == sym)
92 count++;
93
94 // Columns
95 for (int c = 0; c < columns; ++c)
96 for (int r = 0; r <= rows - 3; ++r)
97 if (board[r][c] == sym && board[r + 1][c] == sym && board[r + 2][c] == sym)
98 count++;
99
100 // Main diagonals
101 for (int r = 0; r <= rows - 3; ++r)
102 for (int c = 0; c <= columns - 3; ++c)
103 if (board[r][c] == sym && board[r + 1][c + 1] == sym && board[r + 2][c + 2] == sym)
104 count++;
105
106 // Anti-diagonals
107 for (int r = 2; r < rows; ++r)
108 for (int c = 0; c <= columns - 3; ++c)
109 if (board[r][c] == sym && board[r - 1][c + 1] == sym && board[r - 2][c + 2] == sym)
110 count++;

193Move<char>* Big_XO_UI::get_move(Player<char>* player) {
194 Big_XO_Board* xb = dynamic_cast<Big_XO_Board*>(player->get_board_ptr());
195 if (xb && xb->get_n_moves() >= 24) {
196 cout << "Game over! No more moves allowed.\n";
197 return nullptr;
198 }
199
200 // ---------------- HUMAN INPUT ----------------
201 if (player->get_type() == PlayerType::HUMAN) {
202 int x, y;
203 cout << player->get_name() << " (" << player->get_symbol() << ") enter your move x y: ";
204 cin >> x >> y;
205 return new Move<char>(x, y, player->get_symbol());
206 }
207
208 // ---------------- AI LOGIC ----------------
209 Board<char>* b = player->get_board_ptr();
210 auto mat = b->get_board_matrix();
211 char AI = player->get_symbol();
212 char OP = (AI == 'X' ? 'O' : 'X');
213
214 int bestX = -1, bestY = -1, bestScore = -1;
215
216 auto countSeqAt = [&](int i, int j, char sym) -> int {
217 int cnt = 0;
218 int dx[] = {1,0,1,1};
219 int dy[] = {0,1,1,-1};
220
221 for (int dir = 0; dir < 4; dir++) {
222 int c = 1;
223 int nx = i + dx[dir], ny = j + dy[dir];
224
225 while (nx >= 0 && nx < 5 && ny >= 0 && ny < 5 && mat[nx][ny] == sym) {
226 c++; nx += dx[dir]; ny += dy[dir];
227 }
228
229 nx = i - dx[dir]; ny = j - dy[dir];
230 while (nx >= 0 && nx < 5 && ny >= 0 && ny < 5 && mat[nx][ny] == sym) {
231 c++; nx -= dx[dir]; ny -= dy[dir];
232 }
233
234 if (c >= 3) cnt++;
235 }
236 return cnt;
237 };
238
239 for (int i = 0; i < 5; i++)
240 for (int j = 0; j < 5; j++)
241 if (mat[i][j] == '.') {
242 mat[i][j] = AI;
243 int scoreAI = countSeqAt(i, j, AI);
244 mat[i][j] = '.';
245
246 mat[i][j] = OP;
247 int scoreOP = countSeqAt(i, j, OP);
248 mat[i][j] = '.';
249
250 int totalScore = scoreAI*2 + scoreOP;
251 if (totalScore > bestScore) { bestScore = totalScore; bestX = i; bestY = j; }
252 }
253
254 if (bestX == -1) {
255 if (mat[2][2] == '.') { bestX = 2; bestY = 2; }
256 else {
257 for (int i = 0; i < 5; i++)
258 for (int j = 0; j < 5; j++)
259 if (mat[i][j] == '.') { bestX = i; bestY = j; goto RETURN_MOVE; }
260 }
261 }
262
263RETURN_MOVE:
264 cout << "AI plays at: " << bestX << " " << bestY << "\n";
265 return new Move<char>(bestX, bestY, AI);
