22bool Numerical_Board::update_board(Move<int>* move) {
23 int x = move->get_x();
24 int y = move->get_y();
25 int num = move->get_symbol();
26
27 if (x < 0 || x >= 3 || y < 0 || y >= 3)
28 return false;
29
30 // Cell must be empty and number must not be reused
31 if (board[x][y] != 0 || used_numbers.count(num))
32 return false;
33
34 board[x][y] = num;
35 used_numbers.insert(num);
36 n_moves++;
37 return true;
38}

40bool Numerical_Board::is_win(Player<int>* player) {
41
42 // Check rows and columns

44 if ((board[i][0] + board[i][1] + board[i][2]) == 15 && board[i][0] != 0 && board[i][1] != 0 && board[i][2] != 0) return true;
45 if ((board[0][i] + board[1][i] + board[2][i]) == 15 && board[0][i] != 0 && board[1][i] != 0 && board[2][i] != 0) return true;
46 }
47
48 // diagonals
49 if ((board[0][0] + board[1][1] + board[2][2]) == 15 && board[0][0] != 0 && board[1][1] != 0 && board[2][2] != 0) return true;
50 if ((board[0][2] + board[1][1] + board[2][0]) == 15 && board[0][2] != 0 && board[1][1] != 0 && board[2][0] != 0) return true;
