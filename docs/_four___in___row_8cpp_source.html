19bool FourInRow_Board::update_board(Move<char>* move) {
20 int x = move->get_x();
21 int y = move->get_y();
22 char mark = move->get_symbol();
23
24 // Validate move and apply if valid
25 if (!(x < 0 || x >= rows || y < 0 || y >= columns) &&
26 (board[x][y] == blank_symbol || mark == 0)) {
27
28 if (mark == 0) { // Undo move
29 n_moves--;
30 board[x][y] = blank_symbol;
31 }
32 else { // Apply move
33 n_moves++;
34 board[x][y] = toupper(mark);
35 }
36 return true;
37 }

41bool FourInRow_Board::is_win(Player<char>* player) {
42 const char sym = player->get_symbol();
43
44 auto all_equal = [&](char a, char b, char c, char d) {
45 return a == b && b == c && c == d && a != blank_symbol;
46 };
47 // Rows
48 for (int i = 5; i >=0; --i) {
49 if ((all_equal(board[i][0], board[i][1], board[i][2], board[i][3] ) && board[i][0] == sym) ||
50 (all_equal(board[i][1], board[i][2], board[i][3], board[i][4] ) && board[i][1] == sym) ||
51 (all_equal(board[i][2], board[i][3], board[i][4], board[i][5]) && board[i][2] == sym) ||
52 (all_equal(board[i][3], board[i][4], board[i][5], board[i][6]) && board[i][3] == sym))
53 return true;
54 }
55 // Columns
56 for (int i = 0; i < 7; ++i) {
57 if ((all_equal(board[0][i], board[1][i], board[2][i], board[3][i]) && board[0][i] == sym) ||
58 (all_equal(board[1][i], board[2][i], board[3][i], board[4][i]) && board[1][i] == sym) ||
59 (all_equal(board[2][i], board[3][i], board[4][i], board[5][i]) && board[2][i] == sym))
60 return true;
61 }
62
63 // Main diagonal
64 for (int i = 3; i < 6; ++i) { // Row
65 for (int j = 0; j < 4; ++j) { // Column
66 if (all_equal(board[i][j], board[i - 1][j + 1], board[i - 2][j + 2], board[i - 3][j + 3]) && board[i][j] == sym)
67 return true;
68 }
69 }
70
71 // Anti diagonal
72 for (int i = 3; i < 6; ++i) { // Row
73 for (int j = 3; j < 7; ++j) { // Column
74 if (all_equal(board[i][j], board[i - 1][j - 1], board[i - 2][j - 2], board[i - 3][j - 3]) && board[i][j] == sym)
75 return true;
76 }
77 }
78
79 return false;
80}

104Move<char>* FourInRow_UI::get_move(Player<char>* player) {
105 int x = 1;
106 int y;
107
108 Board<char>* b = player->get_board_ptr(); // get the board
109 auto board = b->get_board_matrix(); // get copy of the original matrix
110 char blank_symbol = '.';
111
112 if (player->get_type() == PlayerType::HUMAN) {
113 while (true) {
114 cout << "\nPlease enter the value of y (0 to 6): ";
115 cin >> y;
116 if (y > 6 || y < 0) { continue; }
117 else { break; }
118 }
119
120 // Find the bottom-most empty cell
121 for (int i = 5; i >= 0; i--) {
122 if (board[i][y] == blank_symbol) {
123 x = i;
124 break;
125 }
126 }
127
128 }
129 else if (player->get_type() == PlayerType::COMPUTER) {
130
131 char me = player->get_symbol();
132 char opp = (me == 'X' ? 'O' : 'X');
133 int R = 6; // Rows
134 int C = 7; // Columns
135
136 int row, col; // For the loops
137
138 // PRIORITY 1 — Win immediately
139
140 for (col = 0; col < C; col++) {
141 row = get_bottom_most_cell(board, col);
142 if (row == -1) continue;
143
144 board[row][col] = me; // Temporary move
145 if (is_winning(board, me))
146 return new Move<char>(row, col, me);
147 board[row][col] = blank_symbol; // Undo
148 }
149 // PRIORITY 2 — Block opponent’s winning move
150
151 for (col = 0; col < C; col++) {
152 row = get_bottom_most_cell(board, col);
153 if (row == -1) continue;
154
155 board[row][col] = opp;
156 if (is_winning(board, opp))
157 return new Move<char>(row, col, me);
158 board[row][col] = blank_symbol;
159 }
160 // PRIORITY 3 — Create a threat (3-in-a-row)
161
162 for (col = 0; col < C; col++) {
163 row = get_bottom_most_cell(board, col);
164 if (row == -1) continue;
165
166 board[row][col] = me;
167 if (create_threat(board, me)) {
168 board[row][col] = blank_symbol;
169 return new Move<char>(row, col, me);
170 }
171 board[row][col] = blank_symbol;
172 }
173 // PRIORITY 4 — Block opponent's threat
174
175 if (create_threat(board, opp)) {
176 // Find all columns that block the threat
177 vector<int> blocking_columns;
178 for (col = 0; col < C; col++) {
179 row = get_bottom_most_cell(board, col);
180 if (row == -1) continue;
181
182 board[row][col] = me;
183 if (!create_threat(board, opp)) {
184 blocking_columns.push_back(col);
185 }
186 board[row][col] = blank_symbol;
187 }
188
189 // If there blocking moves found
190 if (!blocking_columns.empty()) {
191 // Sort blocking columns and prefer rightmost to avoid giving opponent setup
192 int best_col = blocking_columns[0];
193 for (int bc : blocking_columns) {
194 if (bc > best_col) best_col = bc;
195 }
196 row = get_bottom_most_cell(board, best_col);
197 return new Move<char>(row, best_col, me);
198 }
199 }
200 // PRIORITY 5 — Block opponent's 2-in-a-row
201
202 for (col = 0; col < C; col++) {
203 row = get_bottom_most_cell(board, col);
204 if (row == -1) continue;
205
206 board[row][col] = me;
207 if (two_in_row(board, row, col, opp)) {
208 board[row][col] = blank_symbol;
209 return new Move<char>(row, col, me);
210 }
211 board[row][col] = blank_symbol;
212 }
213 // PRIORITY 6 — Prefer center columns
214
215 vector<int> center_preference = { 3, 2, 4, 1, 5, 0, 6 };
216 for (int col : center_preference) {
217 row = get_bottom_most_cell(board, col);
218 if (row != -1) {
219 return new Move<char>(row, col, me);
220 }
221 }
222 }
223 return new Move<char>(x, y, player->get_symbol());
224}
225bool is_winning(const vector<vector<char>>& b, char sym) {
226 int R = 6;
227 int C = 7;
228
229 // Horizontal
230 for (int i = 0; i < R; i++)
231 for (int j = 0; j < C - 3; j++)
232 if (b[i][j] == sym && b[i][j + 1] == sym &&
233 b[i][j + 2] == sym && b[i][j + 3] == sym)
234 return true;
235
236 // Vertical
237 for (int i = 0; i < R - 3; i++)
238 for (int j = 0; j < C; j++)
239 if (b[i][j] == sym && b[i + 1][j] == sym &&
240 b[i + 2][j] == sym && b[i + 3][j] == sym)
241 return true;
242
243 // Main diagonal
244 for (int i = 0; i < R - 3; i++)
245 for (int j = 0; j < C - 3; j++)
246 if (b[i][j] == sym && b[i + 1][j + 1] == sym &&
247 b[i + 2][j + 2] == sym && b[i + 3][j + 3] == sym)
248 return true;
249
250 // Anti diagonal
251 for (int i = 0; i < R - 3; i++)
252 for (int j = 3; j < C; j++)
253 if (b[i][j] == sym && b[i + 1][j - 1] == sym &&
254 b[i + 2][j - 2] == sym && b[i + 3][j - 3] == sym)
255 return true;

257 return false;

260bool create_threat(const vector<vector<char>>& b, char sym) {
261 char blank_symbol = '.';
262
263 int R = 6;
264 int C = 7;
265
266 auto gravity_ok = [&](int r, int c) {
267 return (r == R - 1 || b[r + 1][c] != blank_symbol);
268 };
269
270 // Horizontal
271 for (int r = 0; r < R; r++) {
272 for (int c = 0; c <= C - 4; c++) {
273 int count_sym = 0;
274 int empty_r = -1, empty_c = -1;
275 for (int k = 0; k < 4; k++) {
276 if (b[r][c + k] == sym) count_sym++;
277 else if (b[r][c + k] == blank_symbol) empty_r = r, empty_c = c + k;
278 }
279 if (count_sym == 3 && empty_r != -1 && gravity_ok(empty_r, empty_c)) return true;
280 }
281 }
282
283 // Vertical
284 for (int r = 0; r <= R - 4; r++) {
285 for (int c = 0; c < C; c++) {
286 int count_sym = 0;
287 int empty_r = -1, empty_c = -1;
288 for (int k = 0; k < 4; k++) {
289 if (b[r + k][c] == sym) count_sym++;
290 else if (b[r + k][c] == blank_symbol) empty_r = r + k, empty_c = c;
291 }
292 if (count_sym == 3 && empty_r != -1 && gravity_ok(empty_r, empty_c)) return true;
293 }
294 }
295
296 // Main diagonal
297 for (int r = 0; r <= R - 4; r++) {
298 for (int c = 0; c <= C - 4; c++) {
299 int count_sym = 0;
300 int empty_r = -1, empty_c = -1;
301 for (int k = 0; k < 4; k++) {
302 if (b[r + k][c + k] == sym) count_sym++;
303 else if (b[r + k][c + k] == blank_symbol) empty_r = r + k, empty_c = c + k;
304 }
305 if (count_sym == 3 && empty_r != -1 && gravity_ok(empty_r, empty_c)) return true;
306 }
307 }
308
309 // Anti diagonal
310 for (int r = 0; r <= R - 4; r++) {
311 for (int c = 3; c < C; c++) {
312 int count_sym = 0;
313 int empty_r = -1, empty_c = -1;
314 for (int k = 0; k < 4; k++) {
315 if (b[r + k][c - k] == sym) count_sym++;
316 else if (b[r + k][c - k] == blank_symbol) empty_r = r + k, empty_c = c - k;
317 }
318 if (count_sym == 3 && empty_r != -1 && gravity_ok(empty_r, empty_c)) return true;
319 }
320 }
321
322 return false;

324bool two_in_row(const vector<vector<char>>& b, int r, int c, char opp) {
325 // A defensive way to avoid giving opponent setup
326
327 int R = 6, C = 7;
328
329 // Horizontal
330 int left = 0, right = 0;
331 for (int j = c - 1; j >= 0 && b[r][j] == opp; j--) left++;
332 for (int j = c + 1; j < C && b[r][j] == opp; j++) right++;
333 if (left + right >= 2) return true;
334
335 // Vertical
336 int down = 0;
337 for (int i = r + 1; i < R && b[i][c] == opp; i++) down++;
338 if (down >= 2) return true;
339
340 // Main diagonal
341 int diagonal_down = 0, diagonal_up = 0;
342 for (int i = r + 1, j = c + 1; i < R && j < C && b[i][j] == opp; i++, j++) diagonal_down++;
343 for (int i = r - 1, j = c - 1; i >= 0 && j >= 0 && b[i][j] == opp; i--, j--) diagonal_up++;
344 if (diagonal_down + diagonal_up >= 2) return true;
345
346 // Anti diagonal
347 int anti_down = 0, anti_up = 0;
348 for (int i = r + 1, j = c - 1; i < R && j >= 0 && b[i][j] == opp; i++, j--) anti_down++;
349 for (int i = r - 1, j = c + 1; i >= 0 && j < C && b[i][j] == opp; i--, j++) anti_up++;
350 if (anti_down + anti_up >= 2) return true;
351
352 return false;
353}
